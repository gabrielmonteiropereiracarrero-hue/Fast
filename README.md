            """Inicializa o sistema de armazenamento de diário"""
                    self.storage_file = storage_file

                                    def _load_entries(self):
                                            """Carrega entradas do arquivo JSON"""
                                                    if os.path.exists(self.storage_file):
                                                                try:
                                                                                with open(self.storage_file, 'r', encoding='utf-8') as f:
                                                                                                    return json.load(f)
                                                                                                                except json.JSONDecodeError:
                                                                                                                                return {}
                                                                                                                                        return {}
                                                                                                                                            
                                                                                                                                                def _save_entries(self):
                                                                                                                                                        """Salva entradas no arquivo JSON"""
                                                                                                                                                                with open(self.storage_file, 'w', encoding='utf-8') as f:
                                                                                                                                                                            json.dump(self.entries, f, ensure_ascii=False, indent=2)
                                                                                                                                                                                
                                                                                                                                                                                    def create_entry(self, title, content, tags=None):
                                                                                                                                                                                            """Cria uma nova entrada no diário"""
                                                                                                                                                                                                    entry_id = str(datetime.now().timestamp())
                                                                                                                                                                                                            
                                                                                                                                                                                                                    entry = {
                                                                                                                                                                                                                                "id": entry_id,
                                                                                                                                                                                                                                            "title": title,
                                                                                                                                                                                                                                                        "content": content,
                                                                                                                                                                                                                                                                    "date": datetime.now().isoformat(),
                                                                                                                                                                                                                                                                                "tags": tags or []
                                                                                                                                                                                                                                                                                        }
                                                                                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                                                        self.entries[entry_id] = entry
                                                                                                                                                                                                                                                                                                                self._save_entries()
                                                                                                                                                                                                                                                                                                                        
                                                                                                                                                                                                                                                                                                                                return entry_id

                                                                                                                                                                                                                                                                                                                                                                    return self.entries[entry_id]
                                                                                                                                                                                                                                                                                                                                                                            return None

                                                                                                                                                                                                                                                                                                                                                                                                                                if title:

                                                                                                                                                                                                                                                                                                                                                                                                                                                    if content:
                                                                                                                                                                                                                                                                                                                                                                                                                                                                self.entries[entry_id]["content"] = content
                                                                                                                                                                                                                                                                                                                                                                                                                                                                        if tags is not None:
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    self.entries[entry_id]["tags"] = tags
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    self.entries[entry_id]["updated_at"] = datetime.now().isoformat()
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            self._save_entries()
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            return True
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    def delete_entry(self, entry_id):

                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                del self.entries[entry_id]
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            self._save_entries()
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        return True
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                return False
